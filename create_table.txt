--TABLES

CREATE TABLE ACTOR (
    ACTOR_ID 				NUMBER(9),
    NAME 					VARCHAR2(50) NOT NULL,
    SURNAME					VARCHAR2(50) NOT NULL,
    GENDER					CHAR(1),
    EMAIL					VARCHAR2(50) NOT NULL,
    CITY					VARCHAR2(30) DEFAULT 'LOS ANGELES',
    COUNTRY					CHAR(2) DEFAULT 'US',
    DOB						DATE NOT NULL
);


CREATE TABLE SERIES(
    SERIES_ID				NUMBER(9),
    TITLE					VARCHAR2(100) NOT NULL,
    DESCRIPTION				VARCHAR2(500),
    PROD_YEAR				NUMBER(4) NOT NULL
);

CREATE TABLE EPISODE(
    EPISODE_ID 				NUMBER(9),
    TITLE					VARCHAR2(100) NOT NULL,
    SEASON					NUMBER(3) NOT NULL,
    LENGTH					NUMBER(5),
    SERIES_ID               NUMBER(9) NOT NULL
);

CREATE TABLE CASTING(
    ACTOR_ID				NUMBER(9),
    SERIES_ID				NUMBER(9),
    ROLE 					VARCHAR2(20) NOT NULL
);



CREATE TABLE VIEWER (
    VIEWER_ID 				NUMBER(9),
    NAME					VARCHAR2(50)  NOT NULL,
    SURNAME					VARCHAR2(50) NOT NULL,
    GENDER					CHAR(1),
    EMAIL					VARCHAR2(50) NOT NULL,
    CITY					VARCHAR2(30),
    COUNTRY					CHAR(2) DEFAULT 'US',
    ADDRESS					VARCHAR2(50),
    DOB						DATE
);

CREATE TABLE VIEWERSHIP (
    VIEWER_ID			    NUMBER(9),
    EPISODE_ID				NUMBER(9),
    DOV		                DATE NOT NULL
);
-----------------------------------------------------------

CREATE SEQUENCE seq_episode_id
INCREMENT BY 1;

CREATE SEQUENCE seq_series_id
INCREMENT BY 1;

CREATE SEQUENCE seq_viewer_id
INCREMENT BY 1;
